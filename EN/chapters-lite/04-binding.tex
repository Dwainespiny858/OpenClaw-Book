\chapter{Binding (Route Binding)}

\section{Overview}

A Binding is a rule that routes inbound messages to a specified Agent. In multi-agent mode, Bindings determine which ``brain'' handles which messages.

Config path: \texttt{agents.list[].bindings[]} (or top-level \texttt{bindings[]})

\section{Routing Priority}

\section{Core Concepts}

\begin{center}
\begin{tabular}{ll}
\toprule
\textbf{Term} & \textbf{Description} \\
\midrule
\texttt{agentId} & A ``brain'' (workspace + auth + sessions) \\
\texttt{accountId} & Channel account instance (e.g., WhatsApp \texttt{"personal"} vs \texttt{"biz"}) \\
\texttt{binding} & Routes via (channel, accountId, peer) to agentId \\
\texttt{peer} & Remote identifier (\texttt{kind: "dm"|"group"|"channel"} + \texttt{id}) \\
\bottomrule
\end{tabular}
\end{center}

\section{Configuration Examples}

\subsection{Two WhatsApp Accounts → Two Agents}

\begin{lstlisting}[language=json]
{
  agents: {
    list: [
      { id: "home", default: true, workspace: "~/.openclaw/workspace-home" },
      { id: "work", workspace: "~/.openclaw/workspace-work" }
    ]
  },
  bindings: [
    { agentId: "home", match: { channel: "whatsapp", accountId: "personal" } },
    { agentId: "work", match: { channel: "whatsapp", accountId: "biz" } }
  ]
}
\end{lstlisting}

\subsection{One Number, Split by Sender}

\begin{lstlisting}[language=json]
{
  agents: {
    list: [
      { id: "alex", workspace: "~/.openclaw/workspace-alex" },
      { id: "mia", workspace: "~/.openclaw/workspace-mia" }
    ]
  },
  bindings: [
    { agentId: "alex", match: { channel: "whatsapp", peer: { kind: "dm", id: "+15551230001" } } },
    { agentId: "mia", match: { channel: "whatsapp", peer: { kind: "dm", id: "+15551230002" } } }
  ]
}
\end{lstlisting}

\subsection{Route Specific Group to Specific Agent}

\begin{lstlisting}[language=json]
{
  bindings: [
    {
      agentId: "work",
      match: {
        channel: "whatsapp",
        accountId: "personal",
        peer: { kind: "group", id: "1203630...@g.us" }
      }
    }
  ]
}
\end{lstlisting}
